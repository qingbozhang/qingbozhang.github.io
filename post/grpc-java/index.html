public jnet.grpc.gen.HelloResponse simpleHello(jnet.grpc.gen.HelloRequest request) {
  return blockingUnaryCall(
      getChannel(), getSimpleHelloMethod(), getCallOptions(), request);
}

// 2.
public static <ReqT, RespT> RespT blockingUnaryCall(
      Channel channel, MethodDescriptor<ReqT, RespT> method, CallOptions callOptions, ReqT req) {
    ThreadlessExecutor executor = new ThreadlessExecutor();
    ClientCall<ReqT, RespT> call = channel.newCall(method, callOptions.withExecutor(executor));
    try {
      ListenableFuture<RespT> responseFuture = futureUnaryCall(call, req);
      // 等待并执行完队列的任务，检测 responseFuture 是否完成 即结果是否返回
      while (!responseFuture.isDone()) {
        try {
          executor.waitAndDrain();
        } catch (InterruptedException e) {
          Thread.currentThread().interrupt();
          throw Status.CANCELLED
              .withDescription("Call was interrupted")
              .withCause(e)
              .asRuntimeException();
        }
      }
      // 判断异常并返回结果
      return getUnchecked(responseFuture);
    } catch (RuntimeException e) {
      throw cancelThrow(call, e);
    } catch (Error e) {
      throw cancelThrow(call, e);
    }
  }
// 3.

/**
* 任务队列。只保留关键代码
*/
private static final class ThreadlessExecutor extends ConcurrentLinkedQueue<Runnable>
      implements Executor {
    private static final Logger log = Logger.getLogger(ThreadlessExecutor.class.getName());

    private volatile Thread waiter;

    // Non private to avoid synthetic class
    ThreadlessExecutor() {}

    /**
     * Waits until there is a Runnable, then executes it and all queued Runnables after it.
     * Must only be called by one thread at a time.
     */
    public void waitAndDrain() throws InterruptedException {
      final Thread currentThread = Thread.currentThread();
      throwIfInterrupted(currentThread);
      // 首先在队列中获取一个任务
      Runnable runnable = poll();
      // 如果队列为空则挂起线程，等待被唤醒
      if (runnable == null) {
        waiter = currentThread;
        try {
          while ((runnable = poll()) == null) {
            // 如果队列为空 park 客户端调用线程，即调用 simpleHello() 的线程。
            LockSupport.park(this);
            // 检查是否中断
            throwIfInterrupted(currentThread);
          }
        } finally {
          // 释放 waiter
          waiter = null;
        }
      }
      do {
        try {
          // 在当前线程执行任务，直到队列被消耗空
          runnable.run();
        } catch (Throwable t) {
          log.log(Level.WARNING, "Runnable threw exception", t);
        }
      } while ((runnable = poll()) != null);
    }

    @Override
    public void execute(Runnable runnable) {
      add(runnable);
      // 添加了任务
      LockSupport.unpark(waiter); // no-op if null
    }
  }

ThreadlessExecutor 继承了 ConcurrentLinkedQueue 并实现了 Executor 接口可以调用 execute 向其提交任务，任务提交后被添加到了队列。

首先确定基本思想

  • unary 调用是发出请求并等待服务端回应
  • 客户端服务端的交互是异步的，但是 unary 对外表现起来是同步调用。所以在内部需要异步转同步 很自然我们想到了 Future

回味一下上面这一些代码，在代码块 2的第6行我们果然看到了 Future。 其总体过程就是

  1. 创建 ThreadlessExecutor 执行队列里的任务。
  2. 创建 ClientCall （with第1步的executor）。
  3. 调用服务端并返回 ListenableFuture。
  4. 等待 Future 完成并执行完队列的任务。
  5. 检查异常并返回结果。

看到这里简单粗暴的想法马上就来了，直接 Future.get 阻塞调用等待结果返回不就好了吗？为什么要有 ThreadlessExecutor 呢？ 评感觉也能知道 2、3 是重点。接下来我们会细细研究。
